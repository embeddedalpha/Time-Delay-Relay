119void GPIO_Interrupt_Setup(GPIO_TypeDef *Port, int pin, int edge_select, uint32_t priority, void (*attach_ISR)) {
120 int port_data = PORT_TO_INDEX(Port);
121
122 if (port_data < 0 || pin < 0 || pin > 15) return; // Validate inputs
123
124 // Map GPIO port to EXTI line
125 SYSCFG->EXTICR[pin / 4] |= port_data << ((pin % 4) * 4);
126
127 // Enable EXTI interrupt mask
128 EXTI->IMR |= (1 << pin);
129
130 // Configure rising/falling edge triggers
131 EXTI->RTSR &= ~(1 << pin); // Clear rising edge trigger
132 EXTI->FTSR &= ~(1 << pin); // Clear falling edge trigger
133 switch (edge_select) {
134 case 0:
135 EXTI->RTSR |= (1 << pin);
136 break;
137 case 1:
138 EXTI->FTSR |= (1 << pin);
139 break;
140 case 2:
141 EXTI->RTSR |= (1 << pin);
142 EXTI->FTSR |= (1 << pin);
143 break;
144 }
145
146 // Register the callback function
147 EXTI_ISR[pin] = attach_ISR;
148
149 // Configure NVIC for the EXTI line
150 IRQn_Type irq = (pin <= 4) ? (IRQn_Type)(EXTI0_IRQn + pin) :
151 (pin <= 9) ? EXTI9_5_IRQn : EXTI15_10_IRQn;
152
153 NVIC_SetPriority(irq, priority); // Set interrupt priority
154 NVIC_EnableIRQ(irq); // Enable NVIC interrupt
155}

198GPIO_Status GPIO_Pin_Init(GPIO_TypeDef *Port, uint8_t pin, uint8_t mode, uint8_t output_type, uint8_t speed, uint8_t pull, uint8_t alternate_function) {
199 if (pin > 15 || mode > 3 || speed > 3 || pull > 2) return GPIO_INVALID_PIN;
200
201 GPIO_Clock_Enable(Port);
202
203 // Reset and set mode
204 Port->MODER &= ~(3 << PIN_POS(pin));
205 Port->MODER |= mode << PIN_POS(pin);
206
207 // Reset and set output type
208 Port->OTYPER &= ~(1 << pin);
209 if (output_type != GPIO_Configuration.Output_Type.None) {
210 Port->OTYPER |= output_type << pin;
211 }
212
213 // Reset and set speed
214 Port->OSPEEDR &= ~(3 << PIN_POS(pin));
215 if (speed != GPIO_Configuration.Speed.None) {
216 Port->OSPEEDR |= speed << PIN_POS(pin);
217 }
218
219 // Reset and set pull-up/pull-down
220 Port->PUPDR &= ~(3 << PIN_POS(pin));
221 if (pull != GPIO_Configuration.Pull.None) {
222 Port->PUPDR |= pull << PIN_POS(pin);
223 }
224
225 // Set alternate function
226 if (mode == GPIO_Configuration.Mode.Alternate_Function) {
227 if (pin < GPIO_AF_SPLIT_POINT) {
228 Port->AFR[0] &= ~(0xF << (pin * 4));
229 Port->AFR[0] |= alternate_function << (pin * 4);
230 } else {
231 Port->AFR[1] &= ~(0xF << ((pin - GPIO_AF_SPLIT_POINT) * 4));
232 Port->AFR[1] |= alternate_function << ((pin - GPIO_AF_SPLIT_POINT) * 4);
233 }
234 }
235
236 return GPIO_SUCCESS;
237}
